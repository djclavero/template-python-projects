MANAGING CONDA

# Version 
conda  --version
python --version

# Update
conda update conda
conda update python

# Installations
conda install python=3.7.0

conda install git
conda list

conda install <package>  /  pip install <package>
conda update  <package>  /  pip install --upgrade <package>
conda remove  <package>  /  pip uninstall <package>


CREATE ENVIROMENT

conda create  -n myenv
conda install -n myenv <package>
conda list    -n my env

# Create env with a specific version of python
conda search python
conda create  -n py37 python=3.7.0  			

# Create env with a specific packet
conda create  -n science python    				
conda install -n science scipy

# Create env with all packets of anaconda (.../anaconda2/envs/py37)
conda create  -n py37 python=3.7.0 anaconda 	

conda list    --explicit > spec-file.txt	
# Clone env		
conda create  --name myenv --file spec-file.txt 
# Install all listed packages from file in myenv 	
conda install --name myenv --file spec-file.txt		 


ACTIVATE ENVIROMENT

# List envs (default env 'base')
conda env list	

conda activate myenv

conda deactivate 

# Delete env
conda remove -n myenv all
